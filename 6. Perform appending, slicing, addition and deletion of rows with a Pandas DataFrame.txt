import pandas as pd

# Create a sample DataFrame
data = {
    'Name': ['Alice', 'Bob', 'Charlie', 'David'],
    'Age': [25, 30, 35, 40],
    'City': ['New York', 'London', 'Paris', 'Tokyo']
}

df = pd.DataFrame(data)
print("Original DataFrame:")
print(df)

# 1Ô∏è‚É£ Append a new row
new_row = {'Name': 'Eve', 'Age': 28, 'City': 'Berlin'}
df = df._append(new_row, ignore_index=True)
print("\nAfter Appending a new row:")
print(df)

# 2Ô∏è‚É£ Slice rows (select specific rows)
sliced_df = df[1:4]   # Select rows from index 1 to 3
print("\nAfter Slicing rows (1 to 3):")
print(sliced_df)

# 3Ô∏è‚É£ Add a new row using loc
df.loc[len(df)] = ['Frank', 32, 'Sydney']
print("\nAfter Adding a new row using loc:")
print(df)

# 4Ô∏è‚É£ Delete a row
df = df.drop(2)   # Delete the row with index 2
print("\nAfter Deleting row with index 2:")
print(df)
