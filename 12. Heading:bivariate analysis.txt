# Import required libraries
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load dataset (make sure diabetes.csv is in the same folder)
df = pd.read_csv("diabetes.csv")

# Display first few rows
print("üîπ First 5 records:")
print(df.head())

# Display dataset info
print("\nüîπ Dataset Info:")
print(df.info())

# --------------------------------------------
# 1Ô∏è‚É£ Glucose vs Outcome
# --------------------------------------------
plt.figure(figsize=(6,4))
sns.boxplot(x="Outcome", y="Glucose", data=df, palette="coolwarm")
plt.title("Glucose vs Outcome")
plt.show()

# --------------------------------------------
# 2Ô∏è‚É£ BMI vs Outcome
# --------------------------------------------
plt.figure(figsize=(6,4))
sns.boxplot(x="Outcome", y="BMI", data=df, palette="viridis")
plt.title("BMI vs Outcome")
plt.show()

# --------------------------------------------
# 3Ô∏è‚É£ Age vs Outcome
# --------------------------------------------
plt.figure(figsize=(6,4))
sns.histplot(data=df, x="Age", hue="Outcome", kde=True, multiple="stack")
plt.title("Age Distribution by Outcome")
plt.show()

# --------------------------------------------
# 4Ô∏è‚É£ Pregnancies vs Outcome
# --------------------------------------------
plt.figure(figsize=(6,4))
sns.boxplot(x="Outcome", y="Pregnancies", data=df, palette="Set2")
plt.title("Pregnancies vs Outcome")
plt.show()

# --------------------------------------------
# 5Ô∏è‚É£ Correlation Heatmap
# --------------------------------------------
plt.figure(figsize=(10,6))
sns.heatmap(df.corr(), annot=True, cmap="coolwarm", fmt=".2f")
plt.title("Correlation Heatmap")
plt.show()

# --------------------------------------------
# 6Ô∏è‚É£ Summary Statistics Grouped by Outcome
# --------------------------------------------
print("\nüîπ Mean values grouped by Outcome:")
print(df.groupby("Outcome").mean())
